function golden(f,l0,r0,e=0.1)
    p=(3-5^0.5)/2
    n=ceil(log(e/abs(l0-r0))/log(p))
    if l0<r0
        l1=l0
        r1=r0
    else
        if l0>r0
            l1=r0
            r1=l0
        else
            println("ERROR:l0=r0")
        end
    end
for i=1:n
        l=l1+p*(r1-l1)
        r=l1+(1-p)*(r1-l1)
        fl=f(l)
        fr=f(r)
        println("l=$l,r=$r,fl=$fl,fr=$fr")
        if fl>fr
            l1=r
            r1=r1
        else
            r1=l
            l1=l1
        end 
    end   
    return n
end
